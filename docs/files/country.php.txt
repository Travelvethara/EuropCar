<?php

$ch = curl_init();

$post_string = 'XML-Request=
<message>
  <serviceRequest serviceCode="getStations">
    <serviceParameters>
      <station countryCode="FR" cityName="AGEN" language="FR"/>
    </serviceParameters>
  </serviceRequest>
</message>&callerCode=22467&password=12012015';

curl_setopt($ch, CURLOPT_URL, "https://applications-ptn.europcar.com/xrs/resxml");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, TRUE);
curl_setopt($ch, CURLOPT_HEADER, FALSE);

curl_setopt($ch, CURLOPT_POST, TRUE);

curl_setopt($ch, CURLOPT_POSTFIELDS, $post_string);

curl_setopt($ch, CURLOPT_HTTPHEADER, array("Content-Type: application/x-www-form-urlencoded" 
));

$response = curl_exec($ch);
print_r(($response)); 

$xml = simplexml_load_string($response);
$json = json_encode($xml);
$array = json_decode($json,TRUE);
echo '<pre>';
print_r(($array));
echo '</pre>';
curl_close($ch);

exit;
